    async function loginUser(username, password) {
        try {
            const response = await fetch(SUPABASE_URL + 'players?username=eq.' + encodeURIComponent(username) + '&password=eq.' + encodeURIComponent(password), {
                headers: { 'apikey': SUPABASE_KEY, 'Authorization': 'Bearer ' + SUPABASE_KEY }
            });
            const data = await response.json();
            return data.length > 0 ? { success: true, user: data[0] } : { success: false, message: "Invalid credentials!" };
        } catch (error) { return { success: false, message: "Connection error!" }; }
    }

    function openInGameShop() {
        isShopOpen = true;
        const pauseText = shopOpenedDuringWave ? ' <span style="color:#ff3333; font-size: clamp(12px, 2vw, 16px);">(GAME PAUSED)</span>' : '';
        const titleEl = document.getElementById("shopTitleText");
        if(titleEl) titleEl.innerHTML = `ARMORY & ITEMS (CP: <span id="shopCP">${cpPoints}</span>)${pauseText}`;

        Object.keys(WEAPONS).forEach(k => {

            const div = document.createElement("div"); div.className = "shop-item";
            div.innerHTML = `<div><strong>${w.name}</strong><br><span style="color: #00ffcc;">Cost: ${w.cost} CP</span></div>

            grid.appendChild(div);
            document.getElementById(`buy-${k}`).addEventListener("click", () => {
                if (cpPoints >= w.cost && !isOwned) { 
                    cpPoints -= w.cost; player.ownedWeapons.push(k);
                    
                    // Cancel reload when buying and equipping a new weapon
                    player.isReloading = false;
                    weaponAmmoState[player.currentWeapon] = player.ammo;
                    
                    player.currentWeapon = k; 
                    weaponAmmoState[k] = w.maxAmmo;
                    player.ammo = weaponAmmoState[k];
                    
                    openInGameShop(); 
                }
            });
        });
        document.getElementById("shopScreen").style.display = "flex";
    }

        if(gameState !== "PLAY") return;

        grid.innerHTML = "";

            grid.appendChild(div);
            document.getElementById(`equip-${k}`).onclick = () => {

                player.isReloading = false;

                player.currentWeapon = k;

    function triggerReload() {
        if (gameState !== "PLAY" || player.isReloading) return;
        const w = WEAPONS[player.currentWeapon];
        if (player.ammo >= w.maxAmmo) return; // Already full
        player.isReloading = true;
        player.reloadEndTime = performance.now() + w.reload;
    }

        if(!c || isNaN(x) || isNaN(y) || isNaN(w) || isNaN(h)) return;
        c.fillStyle = "#d2b48c"; c.fillRect(x, y + 20, w, h - 40); c.fillRect(x, y, w, 20);

        c.fillStyle = "#000"; c.fillRect(x + 25, y + 7, 4, 4); c.fillRect(x + 37, y + 7, 4, 4);
        c.strokeStyle = "#000"; c.beginPath(); c.moveTo(x + 24, y + 18); c.quadraticCurveTo(x + 31, y + 22, x + 38, y + 18); c.stroke();
        c.fillStyle = "#d2b48c"; c.fillRect(x + 12, y + h - 20, 6, 20); c.fillRect(x + w - 18, y + h - 20, 6, 20);
        if (currentWeaponKey && WEAPONS[currentWeaponKey]) WEAPONS[currentWeaponKey].draw(c, x + w, y + 36);
    }

    function update(dt) {
        if (gameState !== "PLAY") return;
        
        // Pause logic for inventory/shop
        if (isInvOpen || (isShopOpen && shopOpenedDuringWave)) {
            player.lastShotTime += dt;
            if (player.isReloading) player.reloadEndTime += dt;
            return; 
        }
        
        const now = performance.now();
        if (player.hp < player.maxHp) player.hp = Math.min(player.maxHp, player.hp + 0.1 * (dt / 1000));
        
        let activeSpeedMod = player.speedMod;
        if (keys["KeyA"] || keys["ArrowLeft"] || touchControls.left) player.vx = -activeSpeedMod;
        else if (keys["KeyD"] || keys["ArrowRight"] || touchControls.right) player.vx = activeSpeedMod;
        else player.vx = 0;
        
        if ((keys["Space"] || keys["KeyW"] || keys["ArrowUp"] || touchControls.jump) && player.grounded) { player.vy = -10; player.grounded = false; }
        player.vy += 0.5; player.x += player.vx; player.y += player.vy;
        if (player.y >= 250) { player.y = 250; player.vy = 0; player.grounded = true; }
        if (player.x < 0) player.x = 0; if (player.x > 750) player.x = 750;

        // Reload Logic
        if (keys["KeyR"] || touchControls.reload) {
            triggerReload();
        }

        if (player.isReloading) {
            if (now >= player.reloadEndTime) {
                player.ammo = WEAPONS[player.currentWeapon].maxAmmo;
                player.isReloading = false;
            }
        }

        // Shooting Logic
        if ((keys["KeyF"] || touchControls.shoot) && !player.isReloading) {
            if (now - player.lastShotTime >= WEAPONS[player.currentWeapon].cd) {
                if (player.ammo > 0) {
                    player.ammo--;
                    player.lastShotTime = now;
                    // Note: Bullet creation logic would go here if provided in the original script
                } else {
                    triggerReload(); // Auto reload on empty
                }
            }
        }
    }

    function draw() {
        if (gameState !== "PLAY") return;
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        ctx.fillStyle = "#a0e0ff"; ctx.fillRect(0, 0, BASE_WIDTH, BASE_HEIGHT); 
        ctx.fillStyle = "#87ceeb"; ctx.fillRect(0, 260, BASE_WIDTH, 140); 
        ctx.fillStyle = "#e0c090"; ctx.fillRect(0, 335, BASE_WIDTH, 65); 
        drawTungTungCharacter(ctx, player.x, player.y, player.w, player.h, player.currentWeapon, equippedPerk);
        const ammoStr = player.isReloading ? "RELOADING..." : `${player.ammo}/${WEAPONS[player.currentWeapon].maxAmmo}`;
        const waveStr = waveState === "INTERMISSION" ? `NEXT IN: ${intermissionTimeLeft}s` : (gameMode === "NORMAL" ? `${wave}/67` : `Wave ${wave}`);
        document.getElementById("hudDisplay").innerHTML = `HP: ${Math.ceil(player.hp)}/${player.maxHp}<br>Ammo: ${ammoStr}<br>CP: ${cpPoints}<br>Wave: ${waveStr}<br>TT Bucks: ${tungTungBucks}<br>Kills: ${totalKills}`;
    }

    function drawMenuPose() {
        const pCanvas = document.getElementById("poseCanvas");
        if(!pCanvas) return;
        const pCtx = pCanvas.getContext("2d");
        const displayWidth = pCanvas.clientWidth || 150;
        const displayHeight = pCanvas.clientHeight || 150;
        if(displayWidth === 0 || displayHeight === 0) return;
        pCanvas.width = displayWidth; pCanvas.height = displayHeight;
        pCtx.clearRect(0, 0, displayWidth, displayHeight);
        pCtx.fillStyle = "#1b2a32"; pCtx.fillRect(0, 0, displayWidth, displayHeight);
        const charScale = Math.min(displayWidth / 200, displayHeight / 200);
        if(charScale <= 0) return;
        const charWidth = 50 * charScale; const charHeight = 95 * charScale;
        const charX = (displayWidth - charWidth) / 2; const charY = (displayHeight - charHeight) / 2;
        pCtx.save(); pCtx.scale(charScale, charScale);
        drawTungTungCharacter(pCtx, charX / charScale, charY / charScale, 50, 95, "pistol", equippedPerk);
        pCtx.restore();
    }

    async function initializeGame() {
        loadAudioSettings(); loadGame();
        await tryAutoLogin();
        setTimeout(drawMenuPose, 100);
        applyAudioSettings();
    }

    initializeGame();
    
    let lastTime = performance.now();
    function mainLoop(timestamp) {
        if (!timestamp) timestamp = performance.now();
        let dt = timestamp - lastTime; 
        if (dt > 100) dt = 100; else if (isNaN(dt)) dt = 16;
        lastTime = timestamp;
        update(dt); draw(); requestAnimationFrame(mainLoop);
    }
    requestAnimationFrame(mainLoop);
