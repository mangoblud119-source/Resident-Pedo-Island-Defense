                method: 'PATCH',
                headers: {
                    'apikey': SUPABASE_KEY,
                    'Authorization': 'Bearer ' + SUPABASE_KEY,
                    'Content-Type': 'application/json',
                    'Prefer': 'return=representation'
                },
                body: JSON.stringify({
                    tung_tung_bucks: newBucks
                })
            });

            if (response.ok) {
                showSaveIndicator();
                return true;
            }
            return false;
        } catch (error) {
            console.error("Sync error:", error);
            return false;
        }
    }

    // ======== LOCAL STORAGE FUNCTIONS (for perks/cosmetics) ========
    
    function saveGame() {
        try {
            const saveData = {
                equippedPerk: equippedPerk,
                perks: {}
            };
            
            Object.keys(GLOBAL_PERKS).forEach(key => {
                saveData.perks[key] = GLOBAL_PERKS[key].owned;
            });
            
            localStorage.setItem('residentPedoSave', JSON.stringify(saveData));
        } catch (e) {
            console.warn("Failed to save local data:", e);
        }

        // Sync currency to server if logged in
        if (currentUser) {
            syncToServer(currentUser.username, tungTungBucks);
        }
    }
    
    function loadGame() {
        try {
            const savedData = localStorage.getItem('residentPedoSave');
            if (savedData) {
                const data = JSON.parse(savedData);
                equippedPerk = data.equippedPerk || null;
                
                if (data.perks) {
                    Object.keys(data.perks).forEach(key => {
                        if (GLOBAL_PERKS[key]) {
                            GLOBAL_PERKS[key].owned = data.perks[key];
                        }
                    });
                }
                
                updateWalletsUI();
                return true;
            }
            return false;
        } catch (e) {
            console.warn("Failed to load local data:", e);
            return false;
        }
    }

    function startWave() {
        waveQueue = [];
        enemySpawnTimer = 0;
        waveState = "WAVE";
        bossActive = false;

        if (gameMode === "NORMAL") {
            if (wave === 40 || wave === 50) {
                waveQueue.push("sbe", "sbe", "be", "be", "be");
            } else if (wave === 67) {
                waveQueue.push("sbe", "sbe", "sbe", "sbe", "sbe", "be", "be");
            } else {
                let numEnemies = 5 + (wave - 1); 
                let special = null;
                
                if (wave > 10) {
                    if (wave % 10 === 0) special = "be";
                    else if (wave % 5 === 0) special = "sbe";
                }

                for (let i = 0; i < numEnemies; i++) {
                    if (i === numEnemies - 1 && special) waveQueue.push(special);
                    else waveQueue.push("normal");
                }
            }
        } else if (gameMode === "ENDLESS") {
            let numEnemies = 5 + (wave - 1); 
            for (let i = 0; i < numEnemies; i++) {
                waveQueue.push("normal");
            }
        }
    }

    function openInGameShop() {
        isShopOpen = true;
        const pauseText = shopOpenedDuringWave ? ' <span style="color:#ff3333; font-size:12px;">(GAME PAUSED)</span>' : '';
        const titleEl = document.querySelector("#shopScreen .shop-title");
        titleEl.innerHTML = `WEAPON ARMORY (CP: <span id="shopCP">${cpPoints}</span>)${pauseText}`;
        
        const grid = document.getElementById("shopGrid");
        grid.innerHTML = "";
        Object.keys(WEAPONS).forEach(k => {
            if (k === "pistol") return;
            const w = WEAPONS[k];
            const isOwned = player.ownedWeapons.includes(k);
            const div = document.createElement("div");
            div.className = "shop-item";
            div.innerHTML = `<div><strong>${w.name}</strong><br>Cost: ${w.cost} CP</div>
                             <button id="buy-${k}" ${(cpPoints < w.cost || isOwned) ? 'disabled' : ''}>${isOwned ? 'Owned' : 'Buy'}</button>`;
            grid.appendChild(div);
            document.getElementById(`buy-${k}`).addEventListener("click", () => {
                if (cpPoints >= w.cost && !isOwned) { 
                    cpPoints -= w.cost; player.ownedWeapons.push(k);
                    player.currentWeapon = k; player.ammo = w.maxAmmo;
                    document.getElementById("shopScreen").style.display = "none";
                    isShopOpen = false;
                }
            });
        });
        document.getElementById("shopScreen").style.display = "block";
    }

            const w = WEAPONS[k];
            const isEquipped = (player.currentWeapon === k);
            const div = document.createElement("div");
            div.className = "shop-item";
            div.innerHTML = `<strong>${w.name}</strong><button id="equip-${k}" ${isEquipped ? 'disabled' : ''}>${isEquipped ? 'Equipped' : 'Equip'}</button>`;
            grid.appendChild(div);
            document.getElementById(`equip-${k}`).onclick = () => { 
                player.currentWeapon = k; 
                player.ammo = w.maxAmmo; 
                document.getElementById("invScreen").style.display = "none"; 
                isInvOpen = false; 
            };
        });
        document.getElementById("invScreen").style.display = "block";
    }

    function spawnEnemyFromQueue() {
        if (waveQueue.length === 0) return;
        let type = waveQueue.shift();

        if (type === "sbe") {
            enemies.push({ x: 850, y: 230, w: 65, h: 105, speed: 0.90, hp: 400, maxHp: 400, type: 'sbe', damage: 50, cpPoints: 5, ttb: 1 });
        } else if (type === "be") {
            enemies.push({ x: 850, y: 250, w: 45, h: 85, speed: 2.00, hp: 200, maxHp: 200, type: 'be', damage: 70, cpPoints: 10, ttb: 1, color: '#0f0f0f' });
        } else {
            let baseHp = 3 + (wave - 1);
            enemies.push({ x: 850, y: 250, w: 45, h: 85, speed: 1.50, hp: baseHp, maxHp: baseHp, type: 'normal', damage: 10, cpPoints: 1, ttb: 0 });
        }
    }

    function fireWeapon() {
        const wData = WEAPONS[player.currentWeapon];
        const now = performance.now();
        if (player.isReloading) return;
        if (player.ammo <= 0) { player.isReloading = true; player.reloadEndTime = now + wData.reload; return; }
        if (now - player.lastShotTime >= wData.cd) {
            player.ammo--; player.lastShotTime = now;
            bullets.push({ x: player.x + player.w, y: player.y + 38, vx: 10, vy: 0, damage: wData.damage, weapon: player.currentWeapon });
            
            if (player.currentWeapon === "albert01") {
                player.vy = -3;
                player.x -= 15;
                if (player.x < 0) player.x = 0;
            }
            
            if (player.ammo <= 0) { player.isReloading = true; player.reloadEndTime = now + wData.reload; }
        }
    }

    function checkDeath() {
        if (player.hp <= 0) { 
            gameState = "GAMEOVER"; 
            document.getElementById("gameOverScreen").style.display = "flex"; 
            document.getElementById("quitBtn").style.display = "none";
        }
    }

    function drawTungTungCharacter(c, x, y, w, h, currentWeaponKey, perk) {

        if (perk === "tanky") {

            c.fillStyle = "#111"; c.fillRect(x + 5, y + 25, 10, 15); c.fillRect(x + w - 15, y + 25, 10, 15);
        }
        
        c.fillStyle = "#fff"; c.fillRect(x + 20, y + 4, 10, 10); c.fillRect(x + 32, y + 4, 10, 10);

        if (perk === "shades") {
            c.fillStyle = "#0a0a0a";
            c.fillRect(x + 18, y + 5, 12, 8);
            c.fillRect(x + 32, y + 5, 12, 8);
            c.fillStyle = "#111";

            c.fillStyle = "#1a1a1a";

        }

        if (perk === "cig") {

            c.fillStyle = "#ff4500"; c.fillRect(x + 40, y + 16, 2, 2);
            c.fillStyle = "rgba(200, 200, 200, 0.5)"; c.fillRect(x + 42, y + 12, 3, 3);
        }
        
        if (perk === "cowboy") {
            c.fillStyle = "#8b4513"; c.fillRect(x - 5, y - 2, w + 10, 4);

    function update(dt) {
        if (gameState !== "PLAY") return;
        
        if (isInvOpen || (isShopOpen && shopOpenedDuringWave)) {
            player.lastShotTime += dt;
            if (player.isReloading) player.reloadEndTime += dt;
            return; 
        }

        const now = performance.now();
        
        if (player.hp < player.maxHp) player.hp = Math.min(player.maxHp, player.hp + 0.1 * (dt / 1000));
        if (player.isReloading && now >= player.reloadEndTime) { player.isReloading = false; player.ammo = WEAPONS[player.currentWeapon].maxAmmo; }
        
        let activeSpeedMod = player.speedMod;
        if (player.currentWeapon === "p19") {
            activeSpeedMod *= 0.6; 
        }
        
        if (keys["KeyA"] || keys["ArrowLeft"] || touchControls.left) player.vx = -activeSpeedMod;
        else if (keys["KeyD"] || keys["ArrowRight"] || touchControls.right) player.vx = activeSpeedMod;
        else player.vx = 0;
        
        if ((keys["Space"] || keys["KeyW"] || keys["ArrowUp"] || touchControls.jump) && player.grounded) { player.vy = -10; player.grounded = false; }
        player.vy += 0.5; player.x += player.vx; player.y += player.vy;
        
        if (player.y >= 250) { player.y = 250; player.vy = 0; player.grounded = true; }
        if (player.x < 0) player.x = 0; if (player.x > 750) player.x = 750;
        
        if (keys["KeyE"] || touchControls.shoot) fireWeapon();
        
        if (waveState === "WAVE") {
            enemySpawnTimer += dt;
            if (enemySpawnTimer >= 1500 && waveQueue.length > 0) { 
                spawnEnemyFromQueue(); 
                enemySpawnTimer = 0; 
            }
            if (waveQueue.length === 0 && enemies.length === 0 && !bossActive) {
                if (gameMode === "NORMAL" && wave === 67) {
                    bossActive = true;
                    enemies.push({ x: 650, y: 155, w: 120, h: 180, speed: 0, hp: 2000, maxHp: 2000, type: 'boss', damage: 20, isBoss: true });
                    bossFireballTimer = 0;
                    bossSpikeTimer = 0;
                } else {
                    waveState = "INTERMISSION"; 
                    intermissionTimeLeft = 10; 
                    lastIntermissionTimestamp = now; 
                    cpPoints += 1; tungTungBucks += 1; 
                    saveGame();
                }
            }
        } else if (waveState === "INTERMISSION") {
            if (now - lastIntermissionTimestamp >= 1000) {
                intermissionTimeLeft--; lastIntermissionTimestamp += 1000;
                if (intermissionTimeLeft <= 0) { 
                    wave++; 
                    startWave();
                }
            }
        }

        if (bossActive) {
            let bossRef = enemies.find(e => e.isBoss);
            if (!bossRef) {
                gameState = "VICTORY";
                tungTungBucks += 20;
                if (!GLOBAL_PERKS.shades.owned) {
                    GLOBAL_PERKS.shades.owned = true;
                }
                saveGame();
                updateWalletsUI();
                document.getElementById("victoryScreen").style.display = "flex";
                document.getElementById("quitBtn").style.display = "none";
            } else {
                bossFireballTimer += dt;
                bossSpikeTimer += dt;
                
                if (bossFireballTimer >= 10000) {
                    bossFireballTimer = 0;
                    fireballs.push({ x: bossRef.x, y: 300, w: 20, h: 20, vx: -6, vy: 0, damage: 20 });
                }
                
                if (bossSpikeTimer >= 15000) {
                    bossSpikeTimer = 0;
                    for(let i=0; i<2; i++) {
                        let sx = Math.random() * 550 + 50; 
                        spikes.push({ x: sx, y: 335, w: 20, h: 40, state: 'warning', timer: 3000, activeTimer: 2000, hit: false });
                    }
                }
            }
        }

        for (let i = bullets.length - 1; i >= 0; i--) {
            bullets[i].x += bullets[i].vx; if (bullets[i].x > 850) bullets.splice(i, 1);
        }

        for (let i = fireballs.length - 1; i >= 0; i--) {
            let fb = fireballs[i];
            fb.x += fb.vx;
            if (fb.x < player.x + player.w && fb.x + fb.w > player.x && fb.y < player.y + player.h && fb.y + fb.h > player.y) {
                player.hp -= fb.damage;
                fireballs.splice(i, 1);
                checkDeath();
                continue;
            }
            if (fb.x < -50) fireballs.splice(i, 1);
        }

        for (let i = spikes.length - 1; i >= 0; i--) {
            let sp = spikes[i];
            if (sp.state === 'warning') {
                sp.timer -= dt;
                if (sp.timer <= 0) sp.state = 'active';
            } else if (sp.state === 'active') {
                sp.activeTimer -= dt;
                if (sp.activeTimer <= 0) {
                    spikes.splice(i, 1);
                    continue;
                }
                if (!sp.hit && player.x < sp.x + sp.w && player.x + player.w > sp.x && player.y + player.h > sp.y - sp.h) {
                    player.hp -= 10;
                    sp.hit = true;
                    checkDeath();
                }
            }
        }

        for (let i = floatingTexts.length - 1; i >= 0; i--) {
            floatingTexts[i].y -= 0.5;
            floatingTexts[i].life -= dt;
            if (floatingTexts[i].life <= 0) floatingTexts.splice(i, 1);
        }

        for (let i = enemies.length - 1; i >= 0; i--) {
            let e = enemies[i]; 
            e.x -= e.speed;
            
            if (e.x < player.x + player.w && e.x + e.w > player.x && e.y < player.y + player.h && e.y + e.h > player.y) {
                player.hp -= e.damage; 
                if (e.isBoss) {
                    player.x = e.x - player.w - 20; 
                } else {
                    enemies.splice(i, 1);
                }
                checkDeath();
                continue;
            }
            if (e.x < -150) { enemies.splice(i, 1); continue; }
            
            for (let j = bullets.length - 1; j >= 0; j--) {
                let b = bullets[j];
                if (b.x > e.x && b.x < e.x + e.w && b.y > e.y && b.y < e.y + e.h) {
                    
                    let accCheck = (equippedPerk === "cowboy") ? 1.0 : (WEAPONS[player.currentWeapon].accuracy || 0.95);
                    let isHit = Math.random() <= accCheck;
                    
                    if (!isHit) {
                        floatingTexts.push({ x: e.x + e.w/2 - 15, y: e.y - 10, text: "Miss!", color: "#dddddd", life: 600, maxLife: 600 });
                        bullets.splice(j, 1);
                        break; 
                    }

                    let dmg = b.damage;
                    if (equippedPerk === "cig") dmg *= 1.15;
                    if (equippedPerk === "shades" && Math.random() < 0.3) dmg *= 1.5; 
                    
                    let isCrit = false;
                    if (equippedPerk === "cowboy" && (b.weapon === "judge" || b.weapon === "bigIron") && Math.random() < 0.05) {
                        dmg *= 2;
                        isCrit = true;
                    }
                    
                    e.hp -= dmg; 
                    bullets.splice(j, 1);
                    
                    let textColor = isCrit ? "#ffcc00" : "#ff2222";
                    let textVal = isCrit ? Math.ceil(dmg) + " Crit!" : Math.ceil(dmg);
                    floatingTexts.push({ x: e.x + e.w/2 - 5, y: e.y - 10, text: textVal, color: textColor, life: 600, maxLife: 600 });
                    
                    if (e.hp <= 0) { 
                        if (!e.isBoss) {
                            if (gameMode === "ENDLESS" && wave > 30) {
                                endlessKillCount++;
                                if (endlessKillCount >= 2) {
                                    cpPoints += e.cpPoints;
                                    endlessKillCount -= 2;
                                }
                            } else {
                                cpPoints += e.cpPoints; 
                            }
                            
                            totalKills++; 
                            if (e.type === 'normal' && totalKills % 5 === 0) {
                                tungTungBucks++;
                                saveGame();
                            }
                            if (e.ttb > 0) {
                                tungTungBucks += e.ttb;
                                saveGame();
                            }
                            updateWalletsUI();
                        }
                        enemies.splice(i, 1); 
                        break; 
                    }
                }
            }
        }
    }

    function draw() {
        if (gameState !== "PLAY") return;
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        ctx.fillStyle = "#87ceeb"; ctx.fillRect(0, 260, 800, 140); 
        ctx.fillStyle = "#e0c090"; ctx.fillRect(0, 335, 800, 65); 
        ctx.fillStyle = "#ffffff"; ctx.fillRect(365, 265, 70, 70);
        ctx.fillStyle = "#0000ff"; ctx.fillRect(365, 280, 70, 6); ctx.fillRect(365, 300, 70, 6); ctx.fillRect(365, 320, 70, 6);
        ctx.fillStyle = "#8b4513"; ctx.fillRect(50, 215, 16, 120); ctx.fillRect(734, 215, 16, 120);
        ctx.fillStyle = "#228b22"; ctx.fillRect(20, 195, 76, 20); ctx.fillRect(48, 175, 20, 40); ctx.fillRect(704, 195, 76, 20); ctx.fillRect(732, 175, 20, 40);
        
        enemies.forEach(e => { 
            if (e.isBoss) {
                ctx.fillStyle = "#0a0a0a"; ctx.fillRect(e.x, e.y, e.w, e.h); 
                ctx.fillStyle = "#ff0000"; ctx.fillRect(e.x + 15, e.y + 30, 15, 10); ctx.fillRect(e.x + 45, e.y + 30, 15, 10);
            } else if (e.type === 'be') {
                ctx.fillStyle = e.color; ctx.fillRect(e.x, e.y, e.w, e.h); 
            } else if (e.type === 'sbe') {
                ctx.fillStyle = "#4d5154"; ctx.fillRect(e.x, e.y, e.w, e.h); 
                ctx.fillStyle = "#2b2e30"; ctx.fillRect(e.x + 6, e.y + 12, e.w - 12, e.h - 18); 
                ctx.fillStyle = "#ffcc00"; ctx.fillRect(e.x + 22, e.y + 35, 20, 20); 
                ctx.fillStyle = "#ff4500"; ctx.fillRect(e.x + 18, e.y + 8, 30, 4); 
                ctx.fillStyle = "#708090"; ctx.fillRect(e.x - 10, e.y + 40, 12, 45); 
            } else {
                ctx.fillStyle = "#5a3a22"; ctx.fillRect(e.x, e.y, e.w, e.h); 
            }
            
            ctx.fillStyle = "#000"; ctx.fillRect(e.x-5, e.y-14, e.w+10, 6);
            ctx.fillStyle = e.isBoss ? "#ff0000" : "#0f0"; ctx.fillRect(e.x-4, e.y-13, (e.w+8)*(e.hp/e.maxHp), 4);
        });

        fireballs.forEach(fb => { 
            ctx.fillStyle = "#f00"; ctx.beginPath(); ctx.arc(fb.x + fb.w/2, fb.y + fb.h/2, 12, 0, Math.PI*2); ctx.fill();
            ctx.fillStyle = "#ffa500"; ctx.beginPath(); ctx.arc(fb.x + fb.w/2 + 4, fb.y + fb.h/2, 6, 0, Math.PI*2); ctx.fill();
        });

        spikes.forEach(sp => {
            if (sp.state === 'warning') {
                ctx.fillStyle = "rgba(255, 0, 0, 0.4)"; ctx.fillRect(sp.x, 335 - sp.h, sp.w, sp.h);
                ctx.fillStyle = "#fff"; ctx.font = "bold 20px Arial"; ctx.fillText("!", sp.x + 5, 325);
            } else if (sp.state === 'active') {
                ctx.fillStyle = "#ff0000"; ctx.beginPath();
                ctx.moveTo(sp.x, 335); ctx.lineTo(sp.x + sp.w, 335); ctx.lineTo(sp.x + sp.w/2, 335 - sp.h); ctx.fill();
            }
        });

        bullets.forEach(b => { ctx.fillStyle = "#ff6600"; ctx.fillRect(b.x, b.y, 15, 6); });
        
        drawTungTungCharacter(ctx, player.x, player.y, player.w, player.h, player.currentWeapon, equippedPerk);
        
        floatingTexts.forEach(ft => {
            ctx.fillStyle = ft.color;
            ctx.globalAlpha = Math.max(0, ft.life / ft.maxLife);
            ctx.font = "bold 16px Arial";
            ctx.fillText(ft.text, ft.x, ft.y);
            ctx.globalAlpha = 1.0; 
        });
        
        const ammoStr = player.isReloading ? "RELOADING..." : `${player.ammo}/${WEAPONS[player.currentWeapon].maxAmmo}`;
        const waveStr = waveState === "INTERMISSION" ? `NEXT IN: ${intermissionTimeLeft}s` : (gameMode === "NORMAL" ? `${wave}/67` : `Wave ${wave}`);
        document.getElementById("hudDisplay").innerHTML = `HP: ${Math.ceil(player.hp)}/${player.maxHp}<br>Ammo: ${ammoStr}<br>CP: ${cpPoints}<br>Wave: ${waveStr}<br>TT Bucks: ${tungTungBucks}<br>Kills: ${totalKills}`;
    }

    }

    let lastTime = performance.now();
    function mainLoop(timestamp) {
        let dt = timestamp - lastTime; if (dt > 100) dt = 100; lastTime = timestamp;
        update(dt); draw(); requestAnimationFrame(mainLoop);
    }

    requestAnimationFrame(mainLoop);
